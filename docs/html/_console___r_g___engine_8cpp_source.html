12 bool Console_RG_Engine::execut(const String& newCommand) {

14  String sCommand = strtok(newCommand, " ");
15  Command command = getCommand(sCommand);
16 
17  try {

22  case Command::Load:

37  _help();

39  case Command::Exit:
40  std::cout << "Exiting the program..." << std::endl;
41  toContinue = false;
42  break;
43  case Command::Grayscale:
44  case Command::Monochrome:
45  case Command::Negative:
46  case Command::RotateLeft:
47  case Command::RotateRight:
48  _addInstr(command);
49  break;
50  case Command::Undo:
51  _undo();
52  break;
53  case Command::SwitchSession:
54  _switch();
55  case Command::SessionInfo:
56  _sessionInfo();
57  break;
58  case Command::CollageVertical:
59  case Command::CollageHorizontal:
60  _collage(command);
61  break;
62  }
63  } catch (std::exception e) {
64  std::cout << e.what() << std::endl;
65  }
66 
67  std::cout << std::endl;
68 
69  return toContinue;
70 }

216  if (sCommad == "grayscale") return Command::Grayscale;
217  if (sCommad == "monochrome") return Command::Monochrome;
218  if (sCommad == "negative") return Command::Negative;
219  if (sCommad == "undo") return Command::Undo;
220  if (sCommad == "add") return Command::Add;
221  if (sCommad == "session" && !strcmp(strtok(NULL, " "), "info")) return Command::SessionInfo;
222  if (sCommad == "switch") return Command::SwitchSession;
223  if (sCommad == "rotate") {
224  String direction = strtok(NULL, " ");
225  if(direction == "left") return Command::RotateLeft;
226  if(direction == "right") return Command::RotateRight;
227  }
228  if (sCommad == "collage") {
229  String direction = strtok(NULL, " ");
